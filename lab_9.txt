class Employee{



    private static int numberOfEmployees = 0;
    private static int numberOfActiveEmployees = 0;
    private static int ID_generator=1;
    private int ID;
    private String name;
    private String surname;
    private String login;
    private boolean isActive;
    private List<Employee> employees = new ArrayList<>();



    public Employee(String name, String surname, String login){

        this.name = name;
        this.surname = surname;
        this.login = login;
        this.ID = ID_generator;
        this.isActive = false;
        ID_generator++;
        numberOfEmployees++;
    }



    public void activ(){
        this.isActive = true;
        numberOfActiveEmployees++;
    }
    public void disactiv(){
        this.isActive = false;
        numberOfActiveEmployees--;
    }



    public void add(Employee employee){
        this.employees.add(employee);
    }
    public void remove(Employee employee){
        this.employees.remove(employee);
    }



    public String infoEmployee(){
        return "Name - " +  this.getName() + " Surname - " + this.getSurname() + " Login - "
                + this.getLogin() +  " ID - " + this.getID() + " Is active - " + this.isActive() + "\n"
                + "List of employees it manages : \n" + this.getEmployees();
    }



    public static String howEmployees(){
        return "Total number of employees: " + numberOfEmployees;
    }



    public static String howActiv(){
        return "Total number of active employees: " + numberOfActiveEmployees;
    }



    public static int getNumberOfEmployees() {
        return numberOfEmployees;
    }

    public static int getNumberOfActiveEmployees() {
        return numberOfActiveEmployees;
    }

    public static int getID_generator() {
        return ID_generator;
    }

    public int getID() {
        return ID;
    }

    public String getName() {
        return name;
    }

    public String getSurname() {
        return surname;
    }

    public String getLogin() {
        return login;
    }

    public boolean isActive() {
        return isActive;
    }

    public List<Employee> getEmployees() {
        return employees;
    }

    public void showTable(){
        for(int i = 0; i < employees.size(); i++){
            System.out.println(employees.get(i).getSurname() + employees.get(i).getName() + employees.get(i).getLogin() );
        }
    }

}
