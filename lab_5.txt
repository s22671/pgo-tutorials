class Product{

    private String productName;
    private double productPrice;
    private int productQuantity;


    public Product(String productName,double productPrice, int productQuantity){
        this.productName = productName;
        this.productPrice = productPrice;
        this.productQuantity = productQuantity;
    }
    public String getProductName(){
        return productName;
    }
    public double getProductPrice() {
        return productPrice;
    }
    public int getProductQuantity() {
        return productQuantity;
    }
}

class Category{

    private String categoryName;
    private List<Product> products = new ArrayList<>();

    public Category(List<Product> products){
        this.categoryName = categoryName;
        this.products = products;
    }
    public void showProducts(){
        for(Product product:this.products){
            System.out.println(product.getProductName() + " price is " + product.getProductPrice() + "quantity of it is " + product.getProductQuantity());
        }
    }

    public void add(Product product){
        this.products.add(product);
    }
    public double calculaterPrice(){
        double suma=0;
        for (Product product: this.products){
            suma = suma + product.getProductPrice();

        }
        return  suma;
    }
}

class Basket {
    public List<Product> productList;
    String userName;
    int productsCount;

    public Basket(String user_name) {
        this.productsCount = 0;
        this.productList = new ArrayList<>();
        this.userName = userName;
    }

    public Basket(String user_name, List<Product> products){
        this.productsCount = 0;
        this.userName = userName;
        this.productList = products;
    }

    public boolean addProduct(Product toAdd){
        productsCount++;
        return productList.add(toAdd);
    }

    public boolean removeProduct(Product toRemove){
        productsCount--;
        return productList.remove(toRemove);
    }

    public void pay() {
        int cost = 0;
        for (Product p: productList) {
            cost += p.getProductPrice();
        }
        System.out.println("Total: " + cost);
    }


}
