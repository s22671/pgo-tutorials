class Multiplier {

    public static void main(String args[]) {
        System.out.println(multiply(1,2));
        System.out.println(multiply(1,2,3,4));
        System.out.println(multiply(1.1, 2.2));
        System.out.println(multiply(new BigDecimal(3), new BigDecimal(9)));

        List<Integer> lista = Arrays.asList(1,2,3,4);

        System.out.println(multiply(lista));
    }

    public static int multiply(int a, int b) {
        return a*b;
    }


    public static int  multiply(int a, int b, int c, int d){
    return multiply(a, b) * multiply (c, d);
    }


    public static double multiply(double a, double b) {
        return a*b;
    }


    public static BigDecimal multiply(BigDecimal a, BigDecimal b){
        return a.multiply(b);
    }


    public static Integer multiply(List<Integer> numbers) {
        Integer result = 0;
        for (Integer number : numbers) {
            result = result * number;
        }
        return result;
    }
}

 class Adder{
    public static void main(String args[]) {

        System.out.println(add(1,2));

        List<BigDecimal> numbers = new ArrayList<BigDecimal>();
        numbers.add(new BigDecimal(10));
        numbers.add(new BigDecimal(20));

        System.out.println(add(numbers));
    }

    public static int add(int a, int b){
        return a+b;
    }

    public static BigDecimal add(List<BigDecimal> numbers) {
        BigDecimal result = new BigDecimal("0");
        for (BigDecimal number : numbers) {
            result = result.add(number);
        }

        return result;
    }
}
